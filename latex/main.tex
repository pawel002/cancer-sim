% ── abstract ──────────────────────────────────────────────────────────────
\begin{abstract}\noindent
We benchmark six surrogate strategies for accelerating the simulation of
two-dimensional tumour growth with spatial hypoxia under fractionated
radiotherapy.  Starting from a reaction-diffusion PDE as the ground truth
we compare: (i) a spatially-averaged ODE, (ii) an MLP time-stepper,
(iii)~a Neural ODE with physics-guided correction, (iv) a Physics-Informed
Neural Network, (v) a SuperNet that distils the PINN into a fast ODE
correction, and (vi) a \emph{Trajectory Predictor} (TrajNet)---a
conditional neural network trained on 1\,500 PDE simulations with
randomised parameters that predicts the full mass trajectory in a single
forward pass.  On a canonical $50\!\times\!50$ grid the step-wise
surrogates barely outperform the highly optimised PDE solver, but TrajNet
delivers a \textbf{274$\times$ wall-clock speedup} while maintaining an
RMSE of $1.7\!\times\!10^{-3}$.  On 100 unseen parameter combinations the
median RMSE remains $2.6\!\times\!10^{-3}$, confirming good
generalisation.
All code, trained models and figure-generation scripts are released as
open source.
\end{abstract}

\section{Introduction}
% ══════════════════════════════════════════════════════════════════════════
Solid tumours develop hypoxic (oxygen-deprived) regions as they outgrow
their vasculature \cite{vaupel2004hypoxia,tannock1972relation}.  Because
hypoxic cells are two to three times less radiosensitive than
well-oxygenated cells \cite{hall2006radiobiology}, hypoxia is a major
mechanism of resistance to radiotherapy.  Accurate mathematical modelling
of tumour growth and treatment response is therefore essential for
protocol optimisation \cite{altrock2015mathematics,cristini2010multiscale}.

The standard approach couples logistic proliferation, Fickian diffusion
and a radiation-kill term into a reaction-diffusion PDE
\cite{murray2002mathematical,swanson2000quantitative}.  Integrating this
PDE on a fine mesh over clinically relevant horizons (days to months)
is expensive; the cost becomes prohibitive when thousands of evaluations
are needed---as in Bayesian inference, sensitivity analysis or
digital-twin calibration
\cite{beaumont2002approximate,morris1991factorial,talagrand1987variational}.

Surrogate modelling replaces the PDE solver with a cheaper approximation.
Classical options include spatially-averaged ODEs, which are fast but
discard all spatial information \cite{tracqui2009biophysical}.  Recent
machine-learning methods---MLP time-steppers \cite{lagaris1998artificial},
Neural ODEs \cite{chen2018neural}, PINNs
\cite{raissi2019physics,karniadakis2021physics} and teacher--student
distillation \cite{hinton2015distilling}---offer accuracy improvements but
typically still operate step-by-step and thus inherit much of the PDE's
overhead.

\paragraph{Key observation.}
On a well-optimised $50\!\times\!50$ grid the PDE itself runs in
${\sim}30$\,ms, so step-wise neural surrogates provide \emph{no
meaningful speedup} (0.7--3$\times$ in our experiments).  The decisive
gain comes from \emph{eliminating the time-stepping loop entirely}: a
conditional network trained across many PDE runs predicts the full
trajectory in a single batched forward pass.

\paragraph{Contributions.}
\begin{enumerate}[leftmargin=2em,itemsep=2pt]
  \item We implement six surrogates in a unified framework and show that
        step-wise methods offer negligible speedup on small grids.
  \item We propose a \emph{Trajectory Predictor} (TrajNet) that maps
        physical parameters directly to the full mass trajectory and
        achieves a \textbf{274$\times$ speedup} with an RMSE of
        $1.7\!\times\!10^{-3}$.
  \item We demonstrate that TrajNet generalises across unseen parameter
        combinations (median RMSE $2.6\!\times\!10^{-3}$ over 100 random
        test scenarios).
  \item We release all code, models and reproducible figure generation.
\end{enumerate}

\section{Related Work}
% ══════════════════════════════════════════════════════════════════════════
Reaction-diffusion PDEs of Fisher--KPP type model glioma invasion
\cite{swanson2000quantitative,murray2002mathematical,tracqui2009biophysical};
multiscale extensions are reviewed in \cite{cristini2010multiscale}.
Neural ODEs \cite{chen2018neural,rubanova2019latent} parameterise the
vector field with a network; PINNs \cite{raissi2019physics} minimise the
PDE residual at collocation points.  Karniadakis et al.\
\cite{karniadakis2021physics} and Brunton \& Kutz
\cite{brunton2022data} survey the broader scientific-ML landscape.
Operator-learning methods---FNO \cite{li2020fourier} and DeepONet
\cite{lu2019deeponet}---learn solution maps but require richer training
data; they are a natural extension left for future work.  In the
cancer-modelling domain, Lima et al.\ \cite{lima2021bayesian} use
Gaussian-process surrogates; Yin et al.\ \cite{yin2021non} combine data
assimilation with MRI.  Our TrajNet is most closely related to the
DeepONet paradigm (branch encodes parameters, trunk encodes time) but
uses a simpler concatenation architecture that is easy to train on CPU.

\section{Problem Formulation}\label{sec:problem}
% ══════════════════════════════════════════════════════════════════════════

\subsection{PDE ground-truth model}
Let $\Omega=[0,L]^2$, $u(\bx,t)\in[0,1]$ the normalised cell density,
$K\!=\!1$.  The governing equation is
\begin{equation}\label{eq:pde}
  \partial_t u = D\nabla^2 u + \rho\,u(1-u) - \beta\,R(\bx,t)\,H(\bx)\,u,
  \qquad u\big|_{\partial\Omega}=0,
\end{equation}
with diffusion coefficient~$D$, proliferation rate~$\rho$,
radiosensitivity~$\beta$, dose field $R(\bx,t)=r(t)\,W(\bx)$ (temporal
schedule $\times$ beam mask, here $W\!\equiv\!1$), and hypoxia map
$H(\bx)\in(0,1]$:
\begin{equation}
  H(\bx) = 0.3 + 0.7\,\exp\!\bigl(-\|\bx-\bx_c\|^2/\sigma_H^2\bigr).
\end{equation}

\subsection{ODE surrogate}
Averaging \cref{eq:pde} over~$\Omega$ with Dirichlet BCs yields the
scalar spatial average $U(t)=L^{-2}\!\int_\Omega u\,d\bx$:
\begin{equation}\label{eq:ode}
  \dot U = \rho\,U(1-U) - \beta\,\Heff\,r(t)\,U,
  \qquad \Heff = L^{-2}\!\textstyle\int_\Omega H\,d\bx.
\end{equation}

\subsection{Evaluation protocol}
We split $[0,T]$ into an \textbf{assimilation window} $[0,t_b]$ (used for
training/fitting) and an \textbf{extrapolation window} $(t_b,T]$.
Accuracy is measured by
\begin{equation}\label{eq:rmse}
  \mathrm{RMSE}(\hat U,U_\mathrm{PDE};\mathcal{T})
  = \sqrt{|\mathcal{T}|^{-1}\textstyle\sum_{t_i\in\mathcal{T}}
    (\hat U_i - U_i^\mathrm{PDE})^2}.
\end{equation}

\section{Surrogate Methods}\label{sec:methods}
% ══════════════════════════════════════════════════════════════════════════

\subsection{Step-wise surrogates (baselines)}

\paragraph{MLP time-stepper.}
A two-layer MLP $F_\theta:(U_n,r_n,\dt,\rho,\beta)\mapsto U_{n+1}$
(32~hidden units, ReLU) trained on 150 randomised ODE trajectories via
MSE \cite{hornik1989multilayer,lagaris1998artificial}.

\paragraph{Neural ODE (NODE).}
Extends the ODE with a learned correction
$\dot M = \fode(M,r) + \mathrm{NN}(M,C,r)$ and tracks the
centre-of-mass~$C$ \cite{chen2018neural}.  The physics baseline
$\fode$ from \cref{eq:ode} regularises extrapolation.

\paragraph{PINN.}
A 3-layer MLP $u_\theta(t,x_1,x_2)$ (Tanh$+$Sigmoid) trained to
minimise $w_\mathrm{pde}\mathcal{L}_\mathrm{pde}+
w_\mathrm{ic}\mathcal{L}_\mathrm{ic}+
w_\mathrm{bc}\mathcal{L}_\mathrm{bc}+
w_\mathrm{data}\mathcal{L}_\mathrm{data}$, where the PDE
residual~$\mathcal{L}_\mathrm{pde}$ is evaluated via automatic
differentiation on random collocation points
\cite{raissi2019physics,lu2021deepxde}.

\paragraph{SuperNet.}
Distils PINN knowledge into a fast ODE correction: $U_{n+1}=U_n+\dt
[\fode(U_n,r_n)+g_\phi(U_n,r_n)]$, where $g_\phi$ is a 2-layer MLP
trained on PINN-derived residuals \cite{hinton2015distilling}.

\medskip
All four baselines run \emph{step-by-step}: they advance the state by
one~$\dt$ per call, repeating for all $T/\dt$ steps.  As we show in
\cref{sec:results}, this loop overhead limits their practical speedup on
small grids.

\subsection{Trajectory Predictor (TrajNet)}\label{sec:trajnet}

We propose a fundamentally different approach: train a single network
across many PDE simulations so that at inference time the \emph{entire}
trajectory is obtained in one batched forward pass.

\paragraph{Architecture.}
The surrogate is a conditional function
\begin{equation}\label{eq:trajnet}
  f_\theta : (\rho,\,\beta,\,D,\,A,\,t_s,\,t_e,\;t)
  \;\longmapsto\; U(t),
\end{equation}
where $A$~is the initial-condition amplitude and $[t_s,t_e]$ the
irradiation window.  We implement $f_\theta$ as a 3-layer MLP
(128~hidden units, SiLU activation).  All seven inputs are z-score
normalised.

\paragraph{Data generation.}
We sample 1\,500 parameter vectors from
$\rho\!\in\![0.05,0.6]$,
$\beta\!\in\![0.1,2.0]$,
$D\!\in\![0.002,0.08]$,
$A\!\in\![0.1,1.0]$,
$t_s\!\in\![3,20]$,
$t_e\!-\!t_s\!\in\![3,25]$
and run the PDE for each, recording the normalised mass at 100
uniformly-spaced query times.  Data generation takes ${\sim}43$\,s on a
single CPU core.

\paragraph{Training.}
The 1\,500 trajectories yield $1\,500\times100=150\,000$ input--output
pairs.  The model is trained with Adam (lr$\,=10^{-3}$, cosine annealing)
for 300~epochs with batch size~2\,048, converging to MSE
$8.6\!\times\!10^{-6}$ in about 60\,s.

\paragraph{Inference.}
To predict a trajectory at 100 time points, we form a batch of 100
queries $[(\rho,\beta,D,A,t_s,t_e,t_i)]_{i=1}^{100}$ and evaluate the
network in a single forward pass.  \textbf{Total wall-clock time:
0.12\,ms}---compared to 31.6\,ms for the PDE solver, a
\textbf{274$\times$ speedup}.

\noindent Key observations from \cref{tab:results}:
\begin{itemize}[leftmargin=2em,itemsep=2pt]
  \item \textbf{Step-wise surrogates provide minimal speedup.}
        On the $50\!\times\!50$ grid the PDE runs in 31.6\,ms; all
        step-wise methods still iterate 800 times, so Python/PyTorch
        overhead wipes out any per-step saving.  Only MLP (3$\times$) and
        SuperNet (2$\times$) are faster than the PDE.
  \item \textbf{TrajNet is 274$\times$ faster} because it replaces 800
        sequential steps with a \emph{single} batched forward pass
        through a small MLP.
  \item \textbf{TrajNet is the most accurate} on the canonical scenario
        (RMSE $1.7\!\times\!10^{-3}$), having been trained on
        1\,500 diverse PDE solutions.
  \item \textbf{PINN is the most consistent step-wise method}
        ($\Delta$RMSE${}<7\%$ between windows) thanks to the PDE-residual
        constraint, but is the \emph{slowest} (547\,ms) because it
        evaluates the full 2\,500-point grid at every step.
  \item \textbf{MLP extrapolation drift} confirms that unconstrained
        black-box surrogates are unreliable beyond the training horizon.
\end{itemize}

\section{Discussion}\label{sec:discussion}
% ══════════════════════════════════════════════════════════════════════════

\paragraph{Why step-wise surrogates fail to speed up small-grid PDEs.}
On a $50\!\times\!50$ mesh, \texttt{scipy.ndimage.laplace} executes in
microseconds; the PDE loop completes in ${\sim}30$\,ms.  Any surrogate
that still iterates over 800 steps incurs Python interpreter and PyTorch
dispatch overhead that dominates the per-step computation.  The only way
to achieve large speedups is to \emph{bypass the loop}.

\paragraph{TrajNet as an amortised solver.}
TrajNet shifts all computation to an offline training phase (1\,500 PDE
runs $\approx$43\,s, then 300~epochs $\approx$60\,s) and amortises it
over all future queries.  Even a single query already saves time compared
to running the PDE; in a Monte-Carlo setting with $10^4$ samples the
aggregate saving is ${\sim}5$~minutes of PDE time reduced to
${\sim}1.2$\,s.

\paragraph{When spatial fields are needed.}
TrajNet and the other scalar surrogates predict only the total mass~$U(t)$.
When the full density field is required (e.g.\ for dose planning), the
PINN or PDE must be used.  A natural extension is to predict a small set
of spatial moments---peak density, effective radius, centre of
mass---alongside the mass, adding minimal overhead to TrajNet while
recovering key geometric information.

\paragraph{Limitations.}
\begin{itemize}[leftmargin=2em,itemsep=1pt]
  \item The training set covers a fixed IC shape (Gaussian) and a fixed
        hypoxia map.  Generalising to arbitrary shapes would require
        richer input encodings (e.g.\ shape descriptors or image
        embeddings).
  \item PINN and NODE use fast-benchmark settings (400~epochs, small
        networks).  With longer training they would be more accurate but
        still step-wise.
  \item Extension to 3D or patient-specific geometries is natural but
        would require larger training budgets.
\end{itemize}

\section{Conclusions}
% ══════════════════════════════════════════════════════════════════════════
\begin{enumerate}[leftmargin=2em,itemsep=2pt]
  \item Step-wise neural surrogates (MLP, NODE, PINN, SuperNet) provide
        \emph{negligible speedup} on small 2D grids because the
        time-stepping loop itself is the bottleneck.
  \item The \textbf{Trajectory Predictor} eliminates the loop and
        achieves a \textbf{274$\times$ speedup} with RMSE
        $1.7\!\times\!10^{-3}$, making it suitable for ensemble
        forecasting, Bayesian inference and treatment optimisation.
  \item TrajNet \textbf{generalises robustly} across a six-dimensional
        parameter space (median test RMSE $2.6\!\times\!10^{-3}$ on
        100~unseen scenarios).
  \item Among step-wise methods, the \textbf{PINN} is the most
        extrapolation-resistant thanks to PDE-residual regularisation.
  \item Physics constraints (ODE baseline in NODE and SuperNet; PDE
        residual in PINN) are essential to prevent extrapolation drift
        in learned surrogates.
\end{enumerate}

\medskip\noindent
The modular framework released with this paper allows new engines to be
added by implementing a single \texttt{step()} method and immediately
benefit from the shared training, visualisation and benchmarking
infrastructure.
